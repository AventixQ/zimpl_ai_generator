import re

def convert_response_to_list(response):
    pattern = r"(\d+/\d+|\d+)"
    matches = re.findall(pattern, response)
    result = []
    for match in matches:
        if '/' in match:
            result.append(int(match.split('/')[0]))
        else:
            result.append(int(match))
    
    return result
